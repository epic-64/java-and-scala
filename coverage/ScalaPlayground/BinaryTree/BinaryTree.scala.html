73 case class Tree[A](value: A, left: BinaryTree[A], right: BinaryTree[A]) extends BinaryTree[A] {
74   def insert[B >: A](newValue: B)(using ord: Ordering[B]): BinaryTree[B] =
75     ord.compare(newValue, value) match {
76       case n if n < 0 => Tree(value, left.insert(newValue), right)
77       case n if n > 0 => Tree(value, left, right.insert(newValue))
78       case _          => this
79     }
80 
81   def toList: List[A] = left.toList ++ List(value) ++ right.toList
82 
83   def findShortestPath[B >: A](from: B, to: B)(using ord: Ordering[B]): List[B] = {
84     @tailrec
85     def lowestCommonAncestor(node1: B, node2: B, current: BinaryTree[A]): A =
86       current match
87         case Tree(value, left, right) =>
88           if ord.lt(node1, value) && ord.lt(node2, value) then lowestCommonAncestor(node1, node2, left)
89           else if ord.gt(node1, value) && ord.gt(node2, value) then lowestCommonAncestor(node1, node2, right)
90           else value
91         case EmptyTree                => throw new NoSuchElementException("Nodes not found in the tree")
92 
93     val sharedAncestor = lowestCommonAncestor(from, to, this)
94     val pathFromLocal  = findPathToRoot(from).dropWhile(_ != sharedAncestor).reverse :+ sharedAncestor
95     val pathFromTarget = findPathToRoot(to).dropWhile(_ != sharedAncestor)
96 
97     pathFromLocal.dropRight(1) ++ pathFromTarget.drop(1)
98   }
99 
100   def findPathToRoot[B >: A](target: B)(using ord: Ordering[B]): List[B] =
101     ord.compare(target, value) match {
102       case n if n < 0 => value :: left.findPathToRoot(target)
103       case n if n > 0 => value :: right.findPathToRoot(target)
104       case _          => List(value)
105     }
106 }

115 class TreeFormatter[A](padding: Int = 4) {
116   private def indent(lines: ListBuffer[String], margin: Int): Int = {
117     if (margin >= 0) return margin
118     val spaces = " " * -margin
119     for (i <- lines.indices) lines(i) = spaces + lines(i)
120     0
121   }
122 
123   private def merge(left: ListBuffer[String], right: ListBuffer[String]): ListBuffer[String] = {
124     val minSize = math.min(left.size, right.size)
125     var offset  = 0
126 
127     for (i <- 0 until minSize)
128       offset = math.max(offset, left(i).length + padding - right(i).replaceAll("\\S.*", "").length)
129 
130     indent(right, -indent(left, offset))
131 
132     for (i <- 0 until minSize)
133       left(i) = left(i) + right(i).substring(left(i).length)
134 
135     if (right.size > minSize) {
136       left ++= right.drop(minSize)
137     }
138     left
139   }
140 
141   private def buildLines(node: BinaryTree[A]): ListBuffer[String] = node match {
142     case EmptyTree                => ListBuffer.empty
143     case Tree(value, left, right) =>
144       val leftLines  = buildLines(left)
145       val rightLines = buildLines(right)
146       val lines      = merge(leftLines, rightLines)
147 
148       val half = value.toString.length / 2
149       var i    = half
150 
151       if (lines.nonEmpty) {
152         i = lines.headOption.getOrElse("").indexOf('*') // Marker position
153         val line = (left, right) match {
154           case (EmptyTree, EmptyTree) => " " * i + "┌─┘"
155           case (_, EmptyTree)         => " " * i + "┌─┘"
156           case (EmptyTree, _)         => " " * indent(lines, i - 2) + "└─┐"
157           case (_, _)                 =>
158             val dist = lines.headOption.getOrElse("").length - 1 - i // Calculate distance between roots
159             s"${" " * i}┌${"─" * (dist / 2 - 1)}┴${"─" * ((dist - 1) / 2)}┐"
160         }
161         lines(0) = line
162       }
163 
164       lines.prepend(" " * indent(lines, i - half) + value.toString)
165       lines.prepend(" " * (i + math.max(0, half - i)) + "*") // Add marker for alignment reference
166       lines
167   }
168 
169   def visualize(root: BinaryTree[A]): String = {
170     val lines = buildLines(root)
171     lines.drop(1).mkString("\n") // Drop the alignment marker line
172   }
