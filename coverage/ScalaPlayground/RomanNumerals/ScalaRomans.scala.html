43 class ScalaRomansImperative extends Romans:
44   override def toNumeral(number: Int): String = {
45     val sb        = new StringBuilder
46     var remaining = number
47 
48     for ((value, numeral) <- romanNumerals)
49       while (remaining >= value) {
50         sb.append(numeral)
51         remaining -= value

71 class ScalaRomansIdiomatic extends Romans {
72   def toNumeral(number: Int): String = {
73     @tailrec
74     def convert(remaining: Int, pairs: Vector[(Int, String)], result: String = ""): String =
75       pairs match
76         case (value, numeral) +: _ if remaining >= value => convert(remaining - value, pairs, result + numeral)
77         case _ +: tail                                   => convert(remaining, tail, result)
78         case _                                           => result

81   }
