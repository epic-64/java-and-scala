11 type Floor = Int
12 enum Direction { case Up, Down }

14 case class Person(position: Floor, destination: Floor) {
15   require(position != destination, "source and destination floor cannot be the same")
16 
17   val desiredDirection: Direction = (position, destination) match
18     case _ if destination > position => Up
19     case _ if destination < position => Down
20 
21   def matchesDirection(lift: Lift): Boolean = desiredDirection == lift.direction
22   def isBelow(lift: Lift): Boolean          = position < lift.position
23   def isAbove(lift: Lift): Boolean          = position > lift.position
24 }

26 case class Lift(
27     position: Floor,
28     direction: Direction,
29     people: Queue[Person],
30     capacity: Int
31 ) {
32   def isFull: Boolean    = people.size == capacity
33   def hasRoom: Boolean   = people.size != capacity
34   def hasPeople: Boolean = people.nonEmpty
35   def isEmpty: Boolean   = people.isEmpty
36 
37   def accepts(person: Person): Boolean =
38     hasRoom && person.desiredDirection == direction
39 
40   def nearestPassengerTarget: Option[Floor] =
41     people.filter(_.matchesDirection(this)).map(_.destination).minByOption(floor => Math.abs(floor - position))
42 }

44 case class Building(floors: ListMap[Floor, Queue[Person]]) {
45   def isEmpty: Boolean   = floors.values.forall(_.isEmpty)
46   def hasPeople: Boolean = !isEmpty
47 
48   private def peopleGoing(direction: Direction): List[Person] =
49     floors.values.flatMap(queue => queue.filter(_.desiredDirection == direction)).toList
50 
51   def lowestFloorGoingUp(lift: Lift): Option[Floor] =
52     peopleGoing(Up).filter(_.isBelow(lift)).map(_.position).minOption
53 
54   def highestFloorGoingDown(lift: Lift): Option[Floor] =
55     peopleGoing(Down).filter(_.isAbove(lift)).map(_.position).maxOption
56 
57   def nearestRequestInSameDirection(lift: Lift): Option[Floor] =
58     lift.direction match
59       case Up   => peopleGoing(Up).filter(_.isAbove(lift)).map(_.position).minOption
60       case Down => peopleGoing(Down).filter(_.isBelow(lift)).map(_.position).maxOption
61 }

87 object Dinglemouse {
88   def theLift(queues: Array[Array[Int]], capacity: Int): Array[Int] = {
89     val floors: ListMap[Int, Queue[Person]] =
90       queues.zipWithIndex
91         .map { case (queue, index) =>
92           (index, queue.map(destination => Person(position = index, destination = destination)).to(Queue))
93         }
94         .to(ListMap)
95 
96     val lift     = Lift(position = 0, Direction.Up, people = Queue.empty, capacity)
97     val building = Building(floors)
98 
99     val initialState = State(building = building, lift = lift, stops = List.empty)
100     val finalState   = LiftLogic.simulate(initialState)
101 
102     finalState.stops.toArray
103   }
104 }

106 object LiftLogic {
107   def simulate(initialState: State): State = {
108     val state = initialState.copy(stops = initialState.stops :+ initialState.lift.position)
109 
110     @tailrec
111     def resolve(state: State): State =
112       val newState                 = step(state)
113       val State(building, lift, _) = newState
114       if building.isEmpty && lift.isEmpty && lift.position == 0 then newState
115       else resolve(newState)
116 
117     resolve(state)
118   }
119 
120   private def step(state: State): State = {
121     import state.{building, lift, stops}
122 
123     val validDirection = lift.position match
124       case 0                                  => Up
125       case p if p == building.floors.size - 1 => Down
126       case _                                  => lift.direction
127 
128     // Off-board people who reached their destination
129     val lift2 = lift.copy(
130       direction = validDirection,
131       people = lift.people.filter(_.destination != lift.position)
132     )
133 
134     @tailrec
135     def pickup(lift: Lift, queue: Queue[Person]): (Lift, Queue[Person]) =
136       queue.filter(lift.accepts).dequeueOption match
137         case None            => (lift, queue)
138         case Some(person, _) =>
139           val fullerLift   = lift.copy(people = lift.people.enqueue(person))
140           val emptierQueue = queue.diff(Seq(person))
141           pickup(fullerLift, emptierQueue)
142 
143     // pick up people from the current floor
144     val (lift3, floorQueue) = pickup(lift = lift2, queue = building.floors(lift2.position))
145 
146     // update the building to reflect the updated floor
147     val building2 = building.copy(floors = building.floors.updated(lift3.position, floorQueue))
148 
149     // core task: find the new target and direction
150     val (nextPosition, nextDirection) = getNextPositionAndDirection(building2, lift3)
151 
152     // update lift parameters
153     val lift4 = lift3.copy(nextPosition, nextDirection)
154 
155     // Register the stop. I added the extra condition because of a bug
156     // by which the lift sometimes takes two turns for the very last move 🤔
157     val stops2 = true match
158       case _ if lift3.position != lift4.position => stops :+ lift4.position
159       case _                                     => stops
160 
161     state.copy(building2, lift4, stops2)
162   }
163 
164   private def getNextPositionAndDirection(building: Building, lift: Lift): (Floor, Direction) =
165     List(                                          // Build a list of primary targets
166       lift.nearestPassengerTarget,                 // request from passenger already on the lift
167       building.nearestRequestInSameDirection(lift) // request from people [waiting in AND going to] the same direction
168     ).flatten // turn list of options into list of Integers
169       .minByOption(floor => Math.abs(floor - lift.position)) // get Some floor with the lowest distance, or None
170       .match
171         case Some(floor) => (floor, lift.direction) // return requested floor, keep direction
172         case None        =>                         // otherwise choose a new target
173           lift.direction match
174             case Up   => upwardsNewTarget(building, lift)   // look for people above going downwards
175             case Down => downwardsNewTarget(building, lift) // look for people below going upwards
176 
177   private def downwardsNewTarget(building: Building, lift: Lift): (Floor, Direction) =
178     building.lowestFloorGoingUp(lift) match
179       case Some(lowest) => (lowest, Up)
180       case None         => (building.highestFloorGoingDown(lift).getOrElse(0), Down)
181 
182   private def upwardsNewTarget(building: Building, lift: Lift): (Floor, Direction) =
183     building.highestFloorGoingDown(lift) match
184       case Some(highest) => (highest, Down)
185       case None          => (building.lowestFloorGoingUp(lift).getOrElse(0), Up)
