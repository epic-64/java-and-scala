10 type Floor = Int
11 enum Direction { case Up, Down }

13 case class Person(position: Floor, destination: Floor) {
14   require(position != destination, "source and destination floor cannot be the same")
15 
16   val desiredDirection: Direction = (position, destination) match
17     case _ if destination > position => Up
18     case _ if destination < position => Down
19 
20   def matchesDirection(lift: Lift): Boolean = desiredDirection == lift.direction
21   def isBelow(lift: Lift): Boolean          = position < lift.position
22   def isAbove(lift: Lift): Boolean          = position > lift.position
23 }

25 case class Lift(
26     position: Floor,
27     direction: Direction,
28     people: Queue[Person],
29     capacity: Int
30 ) {
31   def isFull: Boolean    = people.size == capacity
32   def hasRoom: Boolean   = people.size != capacity
33   def hasPeople: Boolean = people.nonEmpty
34   def isEmpty: Boolean   = people.isEmpty
35 
36   def accepts(person: Person): Boolean =
37     hasRoom && person.desiredDirection == direction
38 
39   def nearestPassengerTarget: Option[Floor] =
40     people.filter(_.matchesDirection(this)).map(_.destination).minByOption(floor => Math.abs(floor - position))
41 }

43 case class Building(floors: ListMap[Floor, Queue[Person]]) {
44   def isEmpty: Boolean   = floors.values.forall(_.isEmpty)
45   def hasPeople: Boolean = !isEmpty
46 
47   private def peopleGoing(direction: Direction): List[Person] =
48     floors.values.flatMap(queue => queue.filter(_.desiredDirection == direction)).toList
49 
50   def lowestFloorGoingUp(lift: Lift): Option[Floor] =
51     peopleGoing(Up).filter(_.isBelow(lift)).map(_.position).minOption
52 
53   def highestFloorGoingDown(lift: Lift): Option[Floor] =
54     peopleGoing(Down).filter(_.isAbove(lift)).map(_.position).maxOption
55 
56   def nearestRequestInSameDirection(lift: Lift): Option[Floor] =
57     lift.direction match
58       case Up   => peopleGoing(Up).filter(_.isAbove(lift)).map(_.position).minOption
59       case Down => peopleGoing(Down).filter(_.isBelow(lift)).map(_.position).maxOption
60 }

85 // Excuse the name. Dinglemouse.theLift() is how the function is called in the Codewars test suite
86 object Dinglemouse {
87   def theLift(queues: Array[Array[Int]], capacity: Int): Array[Int] = {
88     val floors: ListMap[Int, Queue[Person]] =
89       queues.zipWithIndex
90         .map { case (queue, index) =>
91           (index, queue.map(destination => Person(position = index, destination = destination)).to(Queue))
92         }
93         .to(ListMap)
94 
95     val lift     = Lift(position = 0, Direction.Up, people = Queue.empty, capacity)
96     val building = Building(floors)
97 
98     val initialState = State(building = building, lift = lift, stops = List.empty)
99     val finalState   = LiftLogic.simulate(initialState)
100 
101     finalState.stops.toArray
102   }
103 }

105 object LiftLogic {
106   def simulate(initialState: State): State = {
107     val state = initialState.copy(stops = initialState.stops :+ initialState.lift.position)
108 
109     @tailrec
110     def resolve(state: State): State =
111       val newState                 = step(state)
112       val State(building, lift, _) = newState
113       if building.isEmpty && lift.isEmpty && lift.position == 0 then newState
114       else resolve(newState)
115 
116     resolve(state)
117   }
118 
119   private def step(state: State): State = {
120     import state.{building, lift, stops}
121 
122     val validDirection = lift.position match
123       case 0                                  => Up
124       case p if p == building.floors.size - 1 => Down
125       case _                                  => lift.direction
126 
127     // Off-board people who reached their destination
128     val lift2 = lift.copy(
129       direction = validDirection,
130       people = lift.people.filter(_.destination != lift.position)
131     )
132 
133     @tailrec
134     def pickup(lift: Lift, queue: Queue[Person]): (Lift, Queue[Person]) =
135       queue.filter(lift.accepts).dequeueOption match
136         case None            => (lift, queue)
137         case Some(person, _) =>
138           val fullerLift   = lift.copy(people = lift.people.enqueue(person))
139           val emptierQueue = queue.diff(Seq(person))
140           pickup(fullerLift, emptierQueue)
141 
142     // pick up people from the current floor
143     val (lift3, floorQueue) = pickup(lift = lift2, queue = building.floors(lift2.position))
144 
145     // update the building to reflect the updated floor
146     val building2 = building.copy(floors = building.floors.updated(lift3.position, floorQueue))
147 
148     // core task: find the new target and direction
149     val (nextPosition, nextDirection) = getNextPositionAndDirection(building2, lift3)
150 
151     // update lift parameters
152     val lift4 = lift3.copy(nextPosition, nextDirection)
153 
154     // Register the stop. I added the extra condition because of a bug
155     // by which the lift sometimes takes two turns for the very last move 🤔
156     val stops2 = true match
157       case _ if lift3.position != lift4.position => stops :+ lift4.position
158       case _                                     => stops
159 
160     state.copy(building2, lift4, stops2)
161   }
162 
163   private def getNextPositionAndDirection(building: Building, lift: Lift): (Floor, Direction) =
164     List(                                          // Build a list of primary targets
165       lift.nearestPassengerTarget,                 // request from passenger already on the lift
166       building.nearestRequestInSameDirection(lift) // request from people [waiting in AND going to] the same direction
167     ).flatten // turn list of options into list of Integers
168       .minByOption(floor => Math.abs(floor - lift.position)) // get Some floor with the lowest distance, or None
169       .match
170         case Some(floor) => (floor, lift.direction) // return requested floor, keep direction
171         case None        =>                         // otherwise choose a new target
172           lift.direction match
173             case Up   => upwardsNewTarget(building, lift)   // look for people above going downwards
174             case Down => downwardsNewTarget(building, lift) // look for people below going upwards
175 
176   private def downwardsNewTarget(building: Building, lift: Lift): (Floor, Direction) =
177     building.lowestFloorGoingUp(lift) match
178       case Some(lowest) => (lowest, Up)
179       case None         => (building.highestFloorGoingDown(lift).getOrElse(0), Down)
180 
181   private def upwardsNewTarget(building: Building, lift: Lift): (Floor, Direction) =
182     building.highestFloorGoingDown(lift) match
183       case Some(highest) => (highest, Down)
184       case None          => (building.lowestFloorGoingUp(lift).getOrElse(0), Up)
