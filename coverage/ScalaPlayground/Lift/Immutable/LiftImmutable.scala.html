13 case class Person(

24   inline def isBelow(lift: Lift): Boolean          = position < lift.position

28 case class Lift(
29     position: Floor,
30     direction: Direction,
31     people: Queue[Person],
32     capacity: Int
33 ) {
34   val isFull: Boolean    = people.size == capacity
35   val hasRoom: Boolean   = people.size != capacity
36   val hasPeople: Boolean = people.nonEmpty
37   val isEmpty: Boolean   = people.isEmpty
38 
39   inline def accepts(person: Person): Boolean =
40     hasRoom && person.desiredDirection == direction
41 
42   inline def nearestPassengerTarget: Option[Floor] =
43     people.filter(_.matchesDirection(this)).map(_.destination).minByOption(floor => Math.abs(floor - position))
44   
45   inline def fixDirection(building: Building): Lift =
46     position match
47       case 0 => copy(direction = Up)
48       case p if p == building.floors.length - 1 => copy(direction = Down)
49       case _ => this
50 
51   inline def dropOff: Lift =
52     copy(people = people.filter(_.destination != position))
53 
54   @tailrec final def pickup(building: Building): (Lift, Building) =
55     val queue = building.floors(position)
56     queue.filter(accepts).dequeueOption match
57       case None            => (this, building) // eventually we reach this case (when there is no one to pick up)
58       case Some(person, _) =>
59         val fullerLift      = copy(people = people.enqueue(person))
60         val emptierQueue    = queue.diff(Seq(person))
61         val emptierBuilding = building.copy(floors = building.floors.updated(position, emptierQueue))
62         fullerLift.pickup(emptierBuilding)
63 
64   inline def align(building: Building): Lift =
65     List(nearestPassengerTarget, building.nearestRequestInSameDirection(this)).flatten
66       .minByOption(floor => Math.abs(floor - position))
67       .match
68         case Some(floor) => copy(position = floor, direction = direction)
69         case None        => // switch direction
70           direction match
71             case Up   =>
72               building.highestFloorGoingDown(this) match
73                 case Some(highest) => copy(highest, Down)
74                 case None          => copy(building.lowestFloorGoingUp(this).getOrElse(0), Up)
75             case Down =>
76               building.lowestFloorGoingUp(this) match
77                 case Some(lowest) => copy(lowest, Up)
78                 case None         => copy(building.highestFloorGoingDown(this).getOrElse(0), Down)
79 }

81 case class Building(
82     floors: Array[Queue[Person]]
83 ) {
84   val isEmpty: Boolean   = floors.forall(_.isEmpty)
85   val hasPeople: Boolean = !isEmpty
86 
87   inline private def peopleGoing(direction: Direction): List[Person] =
88     floors.flatMap(queue => queue.filter(_.desiredDirection == direction)).toList
89 
90   inline def lowestFloorGoingUp(lift: Lift): Option[Floor] =
91     peopleGoing(Up).filter(_.isBelow(lift)).map(_.position).minOption
92 
93   inline def highestFloorGoingDown(lift: Lift): Option[Floor] =
94     peopleGoing(Down).filter(_.isAbove(lift)).map(_.position).maxOption
95 
96   inline def nearestRequestInSameDirection(lift: Lift): Option[Floor] =
97     lift.direction match
98       case Up   => peopleGoing(Up).filter(_.isAbove(lift)).map(_.position).minOption
99       case Down => peopleGoing(Down).filter(_.isBelow(lift)).map(_.position).maxOption
100 }

102 case class LiftSystem(
103     building: Building,
104     lift: Lift,
105     stops: List[Floor]
106 ) {
107   inline private def fixDirection: LiftSystem = copy(lift = lift.fixDirection(building))
108   inline private def dropOff: LiftSystem      = copy(lift = lift.dropOff)
109   inline private def align: LiftSystem        = copy(lift = lift.align(building))
110 
111   inline private def pickup: LiftSystem =
112     val (lift2, building2) = lift.pickup(building)
113     copy(lift = lift2, building = building2)
114 
115   inline private def registerStop: LiftSystem =
116     stops.lastOption match
117       case Some(lastStop) if lastStop == lift.position => this
118       case _                                           => copy(stops = stops :+ lift.position)
119 
120   @tailrec final def resolve: LiftSystem =
121     if building.isEmpty && lift.isEmpty && lift.position == 0
122     then registerStop
123     else registerStop.fixDirection.dropOff.pickup.align.resolve
124 }
