11 type Floor = Int

14 case class Person(position: Floor, destination: Floor) {
15   require(position != destination, "source and destination floor cannot be the same")
16 
17   val desiredDirection: Direction = (position, destination) match
18     case _ if destination > position => Up
19     case _ if destination < position => Down
20 
21   def matchesDirection(lift: Lift): Boolean = desiredDirection == lift.direction
22   def isBelow(lift: Lift): Boolean          = position < lift.position
23   def isAbove(lift: Lift): Boolean          = position > lift.position
24 }

26 case class Lift(
27     position: Floor,
28     direction: Direction,
29     people: Queue[Person],
30     capacity: Int
31 ) {
32   def isFull: Boolean    = people.size == capacity
33   def hasRoom: Boolean   = people.size != capacity
34   def hasPeople: Boolean = people.nonEmpty
35   def isEmpty: Boolean   = people.isEmpty
36 
37   def accepts(person: Person): Boolean =
38     hasRoom && person.desiredDirection == direction
39 
40   def nearestPassengerTarget: Option[Floor] =
41     people.filter(_.matchesDirection(this)).map(_.destination).minByOption(floor => Math.abs(floor - position))
42 }

44 case class Building(floors: ListMap[Floor, Queue[Person]]) {
45   def isEmpty: Boolean   = floors.values.forall(_.isEmpty)
46   def hasPeople: Boolean = !isEmpty
47 
48   private def peopleGoing(direction: Direction): List[Person] =
49     floors.values.flatMap(queue => queue.filter(_.desiredDirection == direction)).toList
50 
51   def lowestFloorGoingUp(lift: Lift): Option[Floor] =
52     peopleGoing(Up).filter(_.isBelow(lift)).map(_.position).minOption
53 
54   def highestFloorGoingDown(lift: Lift): Option[Floor] =
55     peopleGoing(Down).filter(_.isAbove(lift)).map(_.position).maxOption
56 
57   def nearestRequestInSameDirection(lift: Lift): Option[Floor] =
58     lift.direction match
59       case Up   => peopleGoing(Up).filter(_.isAbove(lift)).map(_.position).minOption
60       case Down => peopleGoing(Down).filter(_.isBelow(lift)).map(_.position).maxOption
61 }

89     val floors: ListMap[Int, Queue[Person]] =
90       queues.zipWithIndex

106 object LiftLogic {
107   def simulate(initialState: State): State = {
108     val state = initialState.copy(stops = initialState.stops :+ initialState.lift.position)
109 
110     @tailrec
111     def resolve(state: State): State =
112       val newState                 = step(state)
113       val State(building, lift, _) = newState
114       if building.isEmpty && lift.isEmpty && lift.position == 0 then newState
115       else resolve(newState)
116 
117     resolve(state)
118   }
119 
120   private def step(state: State): State = {
121     import state.{building, lift, stops}
122 
123     val mutableFloors = ListMap.from(building.floors.view.mapValues(mutable.Queue.from))
124     val peopleBuffer  = mutable.Queue.from(lift.people.filter(_.destination != lift.position))
125     val floorQueue    = mutableFloors(lift.position)
126     var direction     = lift.position match
127       case 0                                => Up
128       case p if p == mutableFloors.size - 1 => Down
129       case _                                => lift.direction
130     var position      = lift.position
131     val stopsBuffer   = mutable.ListBuffer.from(stops)
132 
133     // Pickup
134     while peopleBuffer.size < lift.capacity && floorQueue.nonEmpty do
135       val next = floorQueue.dequeue()
136       if next.desiredDirection == direction then peopleBuffer.enqueue(next)
137       else floorQueue.enqueue(next) // put back at end if not accepted
138 
139     val tempLift                      = Lift(position, direction, Queue.from(peopleBuffer), lift.capacity)
140     val tempBuilding                  = Building(mutableFloors.view.mapValues(Queue.from).to(ListMap))
141     val (nextPosition, nextDirection) = getNextPositionAndDirection(tempBuilding, tempLift)
142     position = nextPosition
143     direction = nextDirection
144 
145     if lift.position != position then stopsBuffer += position
146 
147     val finalLift       = Lift(position, direction, Queue.from(peopleBuffer), lift.capacity)
148     val updatedBuilding = Building(mutableFloors.view.mapValues(Queue.from).to(ListMap))
149     state.copy(updatedBuilding, finalLift, stopsBuffer.toList)
150   }
151 
152   private def getNextPositionAndDirection(building: Building, lift: Lift): (Floor, Direction) =
153     List(
154       lift.nearestPassengerTarget,
155       building.nearestRequestInSameDirection(lift)
156     ).flatten
157       .minByOption(floor => Math.abs(floor - lift.position))
158       .match
159         case Some(floor) => (floor, lift.direction)
160         case None        =>
161           lift.direction match
162             case Up   => upwardsNewTarget(building, lift)
163             case Down => downwardsNewTarget(building, lift)
164 
165   private def downwardsNewTarget(building: Building, lift: Lift): (Floor, Direction) =
166     building.lowestFloorGoingUp(lift) match
167       case Some(lowest) => (lowest, Up)
168       case None         => (building.highestFloorGoingDown(lift).getOrElse(0), Down)
169 
170   private def upwardsNewTarget(building: Building, lift: Lift): (Floor, Direction) =
171     building.highestFloorGoingDown(lift) match
172       case Some(highest) => (highest, Down)
173       case None          => (building.lowestFloorGoingUp(lift).getOrElse(0), Up)
